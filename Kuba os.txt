let favorites    = JSON.parse(localStorage.getItem('kubaOS_favorites') || '[]');
let recentPlayed = JSON.parse(localStorage.getItem('kubaOS_recent')    || '[]');
let activeFilter = 'all';
let currentGameName = '';

/* =====================================================
   RECENTLY PLAYED
   ===================================================== */
function recordPlay(gameName) {
  recentPlayed = recentPlayed.filter(n => n !== gameName);
  recentPlayed.unshift(gameName);
  if (recentPlayed.length > 6) recentPlayed = recentPlayed.slice(0, 6);
  localStorage.setItem('kubaOS_recent', JSON.stringify(recentPlayed));
}
function renderRecent() {
  const sec  = document.getElementById('recently-section');
  const grid = document.getElementById('recentGrid');
  grid.innerHTML = '';
  const allGames = [...exclusiveApps, ...libraryGames];
  const games = recentPlayed.map(n => allGames.find(g => g.name === n)).filter(Boolean);
  if (!games.length) { sec.style.display = 'none'; return; }
  sec.style.display = 'block';
  const frag = document.createDocumentFragment();
  games.forEach((g, i) => {
    const c = createCard(g);
    c.style.animationDelay = `${i * 0.04}s`;
    const badge = document.createElement('div');
    badge.className = 'recent-badge';
    badge.textContent = 'Recent';
    badge.style.animationDelay = `${0.15 + i * 0.04}s`;
    c.appendChild(badge);
    frag.appendChild(c);
  });
  grid.appendChild(frag);
}

/* =====================================================
   FAVORITES
   ===================================================== */
function saveFavs()   { localStorage.setItem('kubaOS_favorites', JSON.stringify(favorites)); }
function isFav(name)  { return favorites.includes(name); }
function toggleFav(name) {
  if (isFav(name)) {
    favorites = favorites.filter(f => f !== name);
    showToast('Removed from favorites');
  } else {
    favorites.push(name);
    showToast('❤️ Added to favorites!', 'fav');
  }
  saveFavs();
  document.querySelectorAll(`.fav-btn[data-game="${CSS.escape(name)}"]`).forEach(btn => {
    btn.classList.toggle('active', isFav(name));
    btn.textContent = isFav(name) ? '♥' : '♡';
  });
  renderFavs();
}
function renderFavs() {
  const wrap  = document.getElementById('favorites-section-wrap');
  const grid  = document.getElementById('favoritesGrid');
  const empty = document.getElementById('favorites-empty');
  grid.innerHTML = '';
  const all   = [...exclusiveApps, ...libraryGames];
  const games = favorites.map(n => all.find(g => g.name === n)).filter(Boolean);
  if (!games.length) { wrap.style.display = 'none'; empty.classList.add('show'); return; }
  wrap.style.display = 'block'; empty.classList.remove('show');
  const frag = document.createDocumentFragment();
  games.forEach((g, i) => {
    const c = createCard(g); c.style.animationDelay = `${i * 0.04}s`; frag.appendChild(c);
  });
  grid.appendChild(frag);
}

/* =====================================================
   CARD FACTORY — no innerHTML for security-sensitive data
   ===================================================== */
function createCard(game) {
  const card = document.createElement('div');
  card.className = 'game-card';
  card.dataset.name = game.name.toLowerCase();
  card.dataset.cat  = game.cat || 'other';

  // Thumb wrap
  const thumbWrap = document.createElement('div');
  thumbWrap.className = 'card-thumb-wrap';
  const img = document.createElement('img');
  img.className = 'card-thumb';
  img.src = game.img;
  img.alt = game.name;
  img.loading = 'lazy';
  img.decoding = 'async';
  const shine = document.createElement('div');
  shine.className = 'card-shine';
  thumbWrap.append(img, shine);

  // Info
  const info = document.createElement('div');
  info.className = 'card-info';
  const meta = document.createElement('div');
  const nameEl = document.createElement('div');
  nameEl.className = 'card-name';
  nameEl.textContent = game.name;
  const catEl = document.createElement('div');
  catEl.className = 'card-category';
  catEl.textContent = CATEGORIES[game.cat] || '';
  meta.append(nameEl, catEl);

  const favActive = isFav(game.name);
  const favBtn = document.createElement('button');
  favBtn.className = 'fav-btn' + (favActive ? ' active' : '');
  favBtn.dataset.game = game.name;
  favBtn.title = favActive ? 'Remove from favorites' : 'Add to favorites';
  favBtn.textContent = favActive ? '♥' : '♡';
  info.append(meta, favBtn);

  card.append(thumbWrap, info);

  // Events — URL never stored in DOM attributes
  card.addEventListener('click', e => {
    if (e.target.closest('.fav-btn')) return;
    recordPlay(game.name);
    embedGame(game.url, game.name);  // url only lives in memory
  });
  favBtn.addEventListener('click', e => {
    e.stopPropagation();
    favBtn.classList.add('popping');
    favBtn.addEventListener('animationend', () => favBtn.classList.remove('popping'), { once: true });
    toggleFav(game.name);
  });
  return card;
}

/* =====================================================
   FILTER BAR
   ===================================================== */
function buildFilterBar() {
  const bar = document.getElementById('filter-bar');
  const usedCats = new Set(libraryGames.map(g => g.cat));
  const chips = [['all', '✦ All Games'], ...Object.entries(CATEGORIES).filter(([k]) => usedCats.has(k))];
  const frag = document.createDocumentFragment();
  chips.forEach(([key, label]) => {
    const chip = document.createElement('button');
    chip.className = 'filter-chip' + (key === 'all' ? ' active' : '');
    chip.textContent = label; chip.dataset.cat = key;
    chip.addEventListener('click', () => setFilter(key));
    frag.appendChild(chip);
  });
  bar.appendChild(frag);
}
function setFilter(cat) {
  activeFilter = cat;
  document.querySelectorAll('.filter-chip').forEach(c => c.classList.toggle('active', c.dataset.cat === cat));
  document.querySelectorAll('#gamesGrid .game-card').forEach(card => {
    card.style.display = (cat === 'all' || card.dataset.cat === cat) ? '' : 'none';
  });
  const exVisible = cat === 'all';
  document.getElementById('ex-header').style.display     = exVisible ? '' : 'none';
  document.getElementById('exclusiveGrid').style.display = exVisible ? '' : 'none';
  const anyVisible = [...document.querySelectorAll('#gamesGrid .game-card')].some(c => c.style.display !== 'none');
  document.getElementById('no-results').classList.toggle('show', !anyVisible);
}

/* =====================================================
   RENDER ALL GAMES — use DocumentFragment for perf
   ===================================================== */
function initGames() {
  const exFrag  = document.createDocumentFragment();
  const libFrag = document.createDocumentFragment();

  exclusiveApps.forEach((g, i) => {
    const c = createCard(g); c.style.animationDelay = `${0.1 + i * 0.055}s`; exFrag.appendChild(c);
  });
  libraryGames.forEach((g, i) => {
    const c = createCard(g); c.style.animationDelay = `${0.15 + i * 0.02}s`; libFrag.appendChild(c);
  });

  document.getElementById('exclusiveGrid').appendChild(exFrag);
  document.getElementById('gamesGrid').appendChild(libFrag);
  document.getElementById('lib-count').textContent = libraryGames.length + ' games';
  renderFavs();
  renderRecent();
}

/* =====================================================
   SEARCH — debounced for performance
   ===================================================== */
let searchTimer = null;
const searchInput = document.getElementById('search-input');
searchInput.addEventListener('input', e => {
  clearTimeout(searchTimer);
  searchTimer = setTimeout(() => {
    const q = e.target.value.trim().toLowerCase();
    const allCards = document.querySelectorAll('#exclusiveGrid .game-card, #gamesGrid .game-card');
    let visible = 0;
    allCards.forEach(c => {
      const match = !q || c.dataset.name.includes(q);
      c.style.display = match ? '' : 'none';
      if (match) visible++;
    });
    document.getElementById('no-results').classList.toggle('show', visible === 0 && !!q);
    document.getElementById('favorites-section-wrap').style.display = q ? 'none' : '';
    document.getElementById('recently-section').style.display       = q ? 'none' : '';
  }, 80); // 80ms debounce
});
searchInput.addEventListener('keydown', e => {
  if (e.key === 'Escape') { e.target.value = ''; e.target.dispatchEvent(new Event('input')); }
});

function openRandomGame() {
  const pool = libraryGames.filter(g => activeFilter === 'all' || g.cat === activeFilter);
  if (!pool.length) return;
  const g = pool[Math.floor(Math.random() * pool.length)];
  recordPlay(g.name);
  embedGame(g.url, g.name);
  showToast(`🎲 Playing: ${g.name}`);
}

   GAME WINDOW
   ===================================================== */
let currentIframe = null;
let statusTimer   = null;

function embedGame(url, name) {
  currentGameName = name || 'Game';
  const win     = document.getElementById('app-window');
  const content = document.getElementById('app-content');
  const overlay = document.getElementById('loading-overlay');
  const statusEl= document.getElementById('loading-status');
  const nameEl  = document.getElementById('loading-game-name');

  nameEl.textContent  = name;
  overlay.classList.remove('hidden');
  content.innerHTML = '';

  currentIframe = document.createElement('iframe');
  currentIframe.src = url;
  currentIframe.setAttribute('allow', 'fullscreen; autoplay; clipboard-write');
  currentIframe.setAttribute('sandbox', 'allow-scripts allow-same-origin allow-forms allow-popups allow-presentation allow-orientation-lock');
  currentIframe.style.cssText = 'width:100%;height:100%;border:none;opacity:0;transition:opacity 0.4s ease';
  content.appendChild(currentIframe);

  const statuses = ['Fetching assets…', 'Loading engine…', 'Almost ready…', 'Ready!'];
  let si = 0;
  clearInterval(statusTimer);
  statusTimer = setInterval(() => { if (si < statuses.length) statusEl.textContent = statuses[si++]; }, 650);

  currentIframe.onload = () => {
    clearInterval(statusTimer);
    overlay.classList.add('hidden');
    currentIframe.style.opacity = '1';
  };

  document.getElementById('main-content').classList.add('locked');
  win.classList.remove('closing');
  win.classList.add('open');
  renderRecent();
}
function closeApp() {
  clearInterval(statusTimer);
  const win = document.getElementById('app-window');
  win.classList.add('closing');
  setTimeout(() => {
    win.classList.remove('open', 'closing');
    document.getElementById('app-content').innerHTML = '';
    document.getElementById('main-content').classList.remove('locked');
    document.getElementById('loading-overlay').classList.remove('hidden');
    currentIframe = null;
  }, 380);
}
function refreshGame() {
  if (!currentIframe) return;
  document.getElementById('loading-overlay').classList.remove('hidden');
  currentIframe.style.opacity = '0';
  currentIframe.src += '';
  currentIframe.onload = () => {
    document.getElementById('loading-overlay').classList.add('hidden');
    currentIframe.style.opacity = '1';
  };
}
function fullscreenGame() {
  const f = document.getElementById('app-content').querySelector('iframe');
  if (!f) return;
  (f.requestFullscreen || f.webkitRequestFullscreen || f.mozRequestFullScreen || (()=>{})).call(f);
}

document.addEventListener('keydown', e => {
  if (e.key === 'Escape' && document.getElementById('app-window').classList.contains('open')) closeApp();
});

let settingsOpen = false, cloakOpen = false;
function toggleSettings() {
  settingsOpen = !settingsOpen;
  document.getElementById('settings-modal').style.display = settingsOpen ? 'block' : 'none';
  if (settingsOpen && cloakOpen) { cloakOpen = false; document.getElementById('cloak-panel').style.display = 'none'; }

function toggleCloak() {
  cloakOpen = !cloakOpen;
  document.getElementById('cloak-panel').style.display = cloakOpen ? 'block' : 'none';
  if (cloakOpen && settingsOpen) { settingsOpen = false; document.getElementById('settings-modal').style.display = 'none'; }
}
document.addEventListener('click', e => {
  if (settingsOpen && !document.getElementById('settings-modal').contains(e.target) && e.target.id !== 'settings-btn') {
    settingsOpen = false; document.getElementById('settings-modal').style.display = 'none';
  }
  if (cloakOpen && !document.getElementById('cloak-panel').contains(e.target) && e.target.id !== 'cloak-btn') {
    cloakOpen = false; document.getElementById('cloak-panel').style.display = 'none';
  }
});

function applyCloak(title, iconUrl) {
  document.title = title;
  let link = document.querySelector("link[rel*='icon']");
  if (!link) { link = document.createElement('link'); link.rel = 'icon'; document.head.appendChild(link); }
  link.href = iconUrl;
  showToast(`🛡 Cloaked as: ${title}`);
}
function quickCloak(preset) {
  const p = CLOAK_PRESETS[preset]; if (p) { applyCloak(p.title, p.icon); toggleCloak(); }
}

  const title = document.getElementById('custom-cloak-title').value.trim() || 'Google Classroom';
  const icon  = document.getElementById('custom-cloak-icon').value.trim()  || CLOAK_PRESETS.classroom.icon;

}

async function adminLogin() {
  try {
    await auth.signOut();
    const email = prompt('Admin email:');
    const pw    = prompt('Password:');
    if (!email || !pw) return;
    const u = await auth.signInWithEmailAndPassword(email, pw);
    if (u.user) { isAdmin = true; document.getElementById('banPopup').style.display = 'block'; }
  } catch(e) { alert('Login failed: ' + e.message); }
}
auth.onAuthStateChanged(u => { isAdmin = !!u; });
async function banPlayer() {
  if (!isAdmin) return alert('Admin only');
  const id = document.getElementById('banNumber').value.trim();
  if (!/^\d{6}$/.test(id)) return alert('Invalid ID');
  await db.collection('bans').doc(id).set({ id, timestamp: firebase.firestore.FieldValue.serverTimestamp() });
  alert(`User ${id} banned`);
}
async function unbanPlayer() {
  if (!isAdmin) return alert('Admin only');
  const id = document.getElementById('banNumber').value.trim();
  await db.collection('bans').doc(id).delete();
  alert(`User ${id} unbanned`);
}
async function viewBannedIds() {
  const snap = await db.collection('bans').get();
  alert('Banned: ' + (snap.docs.map(d => d.id).join(', ') || 'None'));
}
